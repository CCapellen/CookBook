% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[35]{min}},
    portion = {\portion{4}}
]
{Bulgarian filled peppers}
    
    \graph
    {% pictures
        %small=pic/cornsoup2,     % small picture
        big=pic/bulgarianpepper % big picture
    }
    
    \introduction{%
        This recipe from Lora is very easy and almost no work. Eat with something else if your feta cheese tolerance is low!
    }
    
    \ingredients{%
    	4 peppers \\
        300-350g Feta cheese \\
        2 eggs \\
        salt and (cayenne) pepper \\
        some oil
    }
    
    \preparation{%
        \step Cut the peppers into half and put them onto a baking dish.
        \step Crumble the cheese and mix it with the eggs, (not too much) salt, pepper and oil.
        \step Put filling in the peppers and bake them at 180 for around 30min till they're brown on top. 
}
    \suggestion[]
    {%
        You can eat the peppers for example with rice and tomato sauce. Or you have small ones as part of a picnic.
    }
    
    
    \hint{  \#easy \#prepareinadvance \#formanypeople \#partyfood \#glutenfree
       
    }
    
\end{recipe}
