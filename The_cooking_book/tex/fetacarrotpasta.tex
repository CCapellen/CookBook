% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[20]{min}},
    portion = {\portion{1-2}}
]
{Feta Carrot Pasta}
    
    \graph
    {% pictures
        %small=pic/fetacarrotpasta2,     % small picture
        big=pic/fetacarrotpasta % big picture
    }
    
    \introduction{%
        A nice and easy dish if you have a grating device. If you have an automated grater, then it's even good for more people
    }
    
    \ingredients{%
        1-2 carrots per person \\
        50g feta cheese per person\\
        some frying oil\\
        1 onion \\
        100g pasta per person \\       
        pepper
    }
    
    \preparation{%
        \step Set water to boil. And boil the pasta
        \step Meanwhile cut the onion into stripes
        \step Grate the carrots 
        \step Fry first the onions, then add the carrots, add some pepper
        \step When they are done to your liking, add the pasta
        \step crumble the cheese and add it
    }
    
    \suggestion[You can write]
    {%
        You can also add green pesto, champignons, ...
        I've tried roast onions, too. It adds a bit of crunch.
    }
    
    
    \hint{%
       
    }
    
\end{recipe}
