% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[25?]{min}},
    portion = {\portion{4+}}
]
{Corn Soup}
    
    \graph
    {% pictures
        %small=pic/cornsoup2,     % small picture
        %big=pic/cornsoup % big picture
    }
    
    \introduction{%
        This is great for gatherings of people. Don't trust the ingredient ratios yet though!
    }
    
    \ingredients{%
    	6 rolls \\
        a can of corn \\
        a pepper \\
        3? tomatoes \\
        a can of pineapple (optional)\\        
        \~200g of grated (gouda) cheese\\ 
        cream \\
        salt and (cayenne) pepper
    }
    
    \preparation{%
        \step Cut the rolls/baguette into half and put it onto a baking dish.
        \step Cut the pepper and the tomatoes into pieces and put them with the corn and the pineapple (maybe not the whole can of that). Add the cream, the spices and the chesse and mix.
        \step Put filling onto the baguettes and bake them. 
}
    \suggestion[]
    {%
        You can prepare the baguettes in advance and bake them when your friends arrive.
    }
    
    
    \hint{  \#easy \#prepareinadvance 
       
    }
    
\end{recipe}
