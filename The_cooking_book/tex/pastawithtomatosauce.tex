\begin{recipe}
[% 
    preparationtime = {\unit[20]{min}},
    %portion = {\portion{1(hungry)-2(add something)}}
]
{Pasta with Tomato Sauce}
    
    \graph
    {% pictures
        small=pic/pastawithtomatosauce,     % small picture
        big=pic/spaghettitomatensosse % big picture
    }
    
    \introduction{%
        This is not as fast as some dishes, but with the right timing, very little work.
    }
    
    \ingredients{%
       	pasta  \\
        tomato sauce\\        
        grated (gouda) cheese
    }
    
    \preparation{%
        \step If you want to leave the kitchen, put the pasta in cold water in a large pot and put it to a moderate heat. Don't use the lid. You can put a wooden spoon on top of the pot as well, which will keep it from boiling over. 
        \step Go do something and after maybe 13 min go check on your pasta. It will take longer than usual and the time depends on the amount of water, speed of the stove, etc.
        \step If you want/need to, warm up the sauce. Mix it with the ready pasta and mix with the cheese.  
}

    }
    
\end{recipe}
