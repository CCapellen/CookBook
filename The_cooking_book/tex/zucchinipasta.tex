\begin{recipe}
[% 
    preparationtime = {\unit[15]{min}},
    portion = {\portion{2-3}}
]
{Pasta with Creamy Zucchini Sauce}
    
    \graph
    {% pictures
            %TODO: better pics
        small=pic/zucchinipasta3,     % small picture
        big=pic/zucchinipasta1 % big picture
    }
    
    \introduction{%
        Quick pasta dish.
    }
    
    \ingredients{%
        \unit[200]{g} pasta (e.g.\ Fussili)\\ %TODO: figure out if this is the right cream to pasta
        1 zucchini \\
        1 small onion\\
        1 clove of garlic\\
        some butter or oil \\
        \unit[100]{ml} cream \\
        salt, pepper \\
        herbs de provence
    }
    
    \preparation{%
        \step Cook the pasta al dente.
        \step While the pasta is cooking, quarter the zucchini lengthwise and dice it.
        \step Dice the onion and finely mince the garlic.
        \step In a pan, saut\'{e} the onion in some butter or oil until translucent. 
        \step Add the zucchini and cook until lightly browned.
        \step Add the garlic and cook until fragrant, make sure not to burn it.
        %TODO \step add some light vinager or lemon juice?
        \step Add the cream and cook until the zucchini is soft and the sauce is slightly reduced. Season with salt, pepper and herbs to taste.
        \step Drain the pasta, add it to the pan and toss until thourougly coated with sauce.
}
    \suggestion[]
    {%
        Herbs de provence is just a mix of savory, marjoram, rosemary, thyme, oregano and lavender. You can use any of these for seasoning.
    }
    
    
    \hint{ Season with salt and pepper to taste whenever you add a new ingredient to the pan. Pairs well with a white wine.
       
    }
    
\end{recipe}
